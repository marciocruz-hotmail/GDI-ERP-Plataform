-- ============================================================
-- BLOCO 9: RENOMEAÇÃO DA PRIMARY KEY
-- Após o rename da tabela, a PK ainda tem nome _NEW.
-- Ajusta para o nome canônico do schema original.
-- ============================================================
PRINT '>>> BLOCO 9: Renomeando a Primary Key para o nome canônico...';

BEGIN TRY
    -- Remove PK com nome canônico se existir (residual improvável, mas defensivo)

    IF EXISTS (

          AND parent_object_id = OBJECT_ID('dbo.gc_movimentos')

    THROW;

-- ============================================================
-- BLOCO 10: RENOMEAÇÃO DAS CONSTRAINTS DEFAULT
-- Renomeia todas as constraints _NEW para os nomes canônicos
-- do schema original. Operação estética/documentacional — não
-- afeta o funcionamento, mas mantém o schema consistente.
-- ============================================================
PRINT '>>> BLOCO 10: Renomeando constraints DEFAULT para nomes canônicos...';

BEGIN TRY
    -- Mapa de renomeação: nome_novo -> nome_canônico
    -- Executado dinamicamente para evitar falhas individuais pararem o bloco inteiro
    DECLARE @old_name NVARCHAR(200), @new_name NVARCHAR(200);
    DECLARE @rename_sql NVARCHAR(500);

    DECLARE cur_rename CURSOR LOCAL FAST_FORWARD FOR
    SELECT dc.name AS old_name,
           REPLACE(dc.name, '_NEW_', '_') AS new_name
    FROM sys.default_constraints dc
    JOIN sys.tables t ON dc.parent_object_id = t.object_id
    WHERE t.name = 'gc_movimentos'
      AND dc.name LIKE 'DF_gc_movimentos_NEW_%';

    OPEN cur_rename;
    FETCH NEXT FROM cur_rename INTO @old_name, @new_name;

    WHILE @@FETCH_STATUS = 0
    BEGIN
        BEGIN TRY
            -- Só renomeia se o nome canônico ainda não existir
            IF NOT EXISTS (
                SELECT 1 FROM sys.default_constraints dc2
                JOIN sys.tables t2 ON dc2.parent_object_id = t2.object_id
                WHERE t2.name = 'gc_movimentos' AND dc2.name = @new_name
            )
            BEGIN
                SET @rename_sql = N'EXEC sp_rename ''' + @old_name + N''', ''' + @new_name + N''', ''OBJECT'';';
                EXEC sp_executesql @rename_sql;
            END
        END TRY
        BEGIN CATCH
            PRINT '  [AVISO] Não foi possível renomear constraint ' + @old_name + ': ' + ERROR_MESSAGE();
        END CATCH

        FETCH NEXT FROM cur_rename INTO @old_name, @new_name;
    END

    CLOSE cur_rename;
    DEALLOCATE cur_rename;

    PRINT '  Renomeação de constraints concluída.';
END TRY
BEGIN CATCH
    PRINT '  [ERRO] Falha no bloco de renomeação de constraints: ' + ERROR_MESSAGE();
    -- Não propaga: renomeação de constraints é cosmética, não bloqueia o schema.
END CATCH
GO

(
    id_atendimento               int            IDENTITY(1,1) NOT NULL,
    concluido                    bit            NOT NULL CONSTRAINT DF_g_atendimentos_concluido DEFAULT (0),
    solicitacao                  nvarchar(100)  NULL,
    descricao                    nvarchar(max)  NULL,
    anotacoes                    nvarchar(max)  NULL,
    privado                      bit            NOT NULL CONSTRAINT DF_g_atendimentos_privado DEFAULT (0),
    enviar_atualizacoes          bit            NOT NULL CONSTRAINT DF_g_atendimentos_enviar_atualizacoes DEFAULT (1),
    email_atualizacoes           nvarchar(max)  NULL,

    /* Opcional: permitir NULL (evita “0 inválido” e facilita FK) */

    param_numero_pedido                 int     NOT NULL CONSTRAINT DF_g_atendimentos_param_numero_pedido DEFAULT (0),
    param_numero_nf                     int     NOT NULL CONSTRAINT DF_g_atendimentos_param_numero_nf DEFAULT (0),
    param_id_produto                    int     NOT NULL CONSTRAINT DF_g_atendimentos_param_id_produto DEFAULT (0),
    param_limite_credito                int     NOT NULL CONSTRAINT DF_g_atendimentos_param_limite_credito DEFAULT (0),
    param_id_vendedor                   int     NOT NULL CONSTRAINT DF_g_atendimentos_param_id_vendedor DEFAULT (0),

    /* Obrigatório (ticket deve ter status e categoria) */
    id_status                    int            NOT NULL,
    id_atendimento_categoria     int            NOT NULL,

    solicitacao_id_usuario       int            NOT NULL,
    solicitacao_datahora         datetime2(0)   NOT NULL,

    responsavel_id_usuario       int            NULL,
    responsavel_id_departamento  int            NULL,

    conclusao_id_usuario         int            NULL,
    conclusao_datahora           datetime2(0)   NULL,

    cancelamento_id_usuario      int            NULL,
    cancelamento_datahora        datetime2(0)   NULL,

    id_coligada                  int            NOT NULL CONSTRAINT DF_g_atendimentos_id_coligada DEFAULT (0),
    id_filial                    int            NOT NULL CONSTRAINT DF_g_atendimentos_id_filial DEFAULT (0),

    datahora_cadastro            datetime2(0)   NOT NULL CONSTRAINT DF_g_atendimentos_datahora_cadastro DEFAULT (SYSDATETIME()),
    id_usuario_cadastro          int            NOT NULL,
    datahora_alteracao           datetime2(0)   NULL,
    id_usuario_alteracao         int            NULL,

    CONSTRAINT PK_g_atendimentos PRIMARY KEY CLUSTERED (id_atendimento)
);
